# MySQL

# mysql 程序的常用命令

show datebases：显示所有数据库

use dbname：选定默认数据库

show tables：显示默认数据库中所有表

\c：放弃正在输入的命令

\h：显示命令清单

\q：退出 mysql 程序

\s：查看 mysql 服务器状态信息

# 数据类型种类

# 1、整形

可以有十进制或十六进制表示（0x 后接相应的数）

# 2、浮点型

有一个数字加一个小数点再接一个数字组成

# 3、字符串

有单引号或双引号括起来的字符或数字，要使用转义字符才能显示特殊符号

# 4、时间和日期值

日期按照 年 - 月 - 日 的顺序

# 5、NULL 值

无类型的值，表示空

# 6、数值列类型

TINYINT：非常小的整数

BIGINT：大整数

SMALLINT：较小的整数

FLOAT：单精度浮点数

MEDIUMINT：中等大小的整数

DOUBLE：双精度浮点数

INT：标准整数

DECIMAL：一个串的浮点数

# 7、字符串列类型

CHAR：定长字符串

ENUM：枚举列，可赋予某个枚举成员

VARCAHR：可变长字符串

SET：集合列，可赋予多个集合成员

BLOB：存储图像文件

TEXT：小文本串

# 常用 SQL 语句

# 创建数据库操作

create datebase 数据库名

# 建立表操作

create table 表名 {

列名 列类型 [<完整性约束>]

…

}

# 显示表结构

describe 表名 / desc 表名

# 删除表

drop table [if exists] tab_name,[table_name]…

# 更改表结构

alter table 表名 action

action 可以是以下语句

add 列名 建表语句 [first | after 列名]：添加一列

add primary key (列名)：添加主键，已存在主键报错

modify 列名 <建表语句> [first | after 列名]: 更改类型和列名称

drop 列名：删除以一列

rename as 新表名：更改表名

# 插入记录操作

INSRT [INTO] < 表名 > [列名] VALUES < 值列表 >

# 更改数据行

UPDATE <表名> SET < 列名 = 更新值 > [WHERE < 更新条件 >]

# 删除记录

DELETE FROM <表名> [WHERE < 删除条件 >]

# 查询操作

SELECT <列名> FROM < 表名 > [WHERE < 查询条件表达式 >] [ORDER BY < 排序的列名 >[ASC 或 DESC]]

# PHP 操作 mysql

# php 访问 mysql 数据库流程

# 连接 mysql 服务器

使用 mysql_connect（）与 MySQL 服务器建立连接

# 选择 mysql 数据库

使用 mysql_select_db（）函数选择 MySQL 数据库服务器上的数据库，并与数据库建立连接

# 执行 sql 语句

使用 mysql_query（）函数执行 SQL 语句

# 关闭结果集

mysql_free_result ($result) 关闭结果集

# 关闭 mysql 服务器

mysql_close（）函数关闭数据库连接
